# 第一章.购物车

# 1.购物车原型分析

1.通过观察发现:一个购物车中都多个购物项
    
2.购物项的属性有: 
  图书id(bookId)  
  图书名称(bookName)   
  图书图片路径(imgPath)  
  图书单价(price)  
  图书数量(count)   
  图书金额(amount)  //金额的多少应该是数量*单价决定的
      
3.购物车
  a.属性有:
    totalAmount(购物车总金额)
    totalCount(购物车商品总数)
    Map集合   -> 存储购物项   应该用Map(存储购物项的Map集合,购物项的bookId作为key,购物项CartItem作为value)
            因为删除,添加某一个商品,购物车中的其他商品不能动,所以map中的key应该是图书的id

# 1.1.购物项类_CartItem

public class CartItem {
    //图书id  因为图书包含在购物项之内
    private Integer bookId;
    //购物项中图书的名称
    private String bookName;
    //购物项中图片的路径
    private String imgPath;
    //购物项中图书的单价
    private Double price = 0.0;
    //购物项中图书的数量
    private Integer count = 0;
    //购物项中图书的金额
    private Double amount = 0.0;

    public CartItem() {
    }

    public CartItem(Integer bookId, String bookName, String imgPath, Double price, Integer count, Double amount) {
        this.bookId = bookId;
        this.bookName = bookName;
        this.imgPath = imgPath;
        this.price = price;
        this.count = count;
        this.amount = amount;
    }

    public Integer getBookId() {
        return bookId;
    }

    public void setBookId(Integer bookId) {
        this.bookId = bookId;
    }

    public String getBookName() {
        return bookName;
    }

    public void setBookName(String bookName) {
        this.bookName = bookName;
    }

    public String getImgPath() {
        return imgPath;
    }

    public void setImgPath(String imgPath) {
        this.imgPath = imgPath;
    }

    public Double getPrice() {
        return price;
    }

    public void setPrice(Double price) {
        this.price = price;
    }

    public Integer getCount() {
        return count;
    }

    public void setCount(Integer count) {
        this.count = count;
    }

    //计算小计金额
    public Double getAmount() {
        this.amount = this.count*this.price;
        return amount;
    }

    public void setAmount(Double amount) {
        this.amount = amount;
    }
}

# 1.2.购物车类_Cart

public class Cart {
    //购物车价格总金额
    private Double totalAmount = 0.0;
    //购物车商品总数量
    private Integer totalCount = 0;
    /*
      由于购物车中需要存储多个购物项
      所以我们定义Map集合,key为图书的id,value为购物项对象来存储购物项对象

    private Map<Integer, CartItem> cartItemMap = new HashMap<>();

    }

        this.totalCount = totalCount;

        return totalAmount;

    }

    public Integer getTotalCount() {
        return totalCount;
    }

    public void setTotalCount(Integer totalCount) {
        this.totalCount = totalCount;
    }

    public Map<Integer, CartItem> getCartItemMap() {
        return cartItemMap;
    }

    public void setCartItemMap(Map<Integer, CartItem> cartItemMap) {
        this.cartItemMap = cartItemMap;
    }
}

# 2.1.CartServlet添加购物车

@WebServlet("/cart")
public class CartServlet extends BaseServlet {
    BookService bookService = new BookServiceImpl();
    public void toCartPage(HttpServletRequest request,HttpServletResponse response) throws Exception {
        processTemplate("cart/cart",request,response);
    }

    /*
       添加购物车
     */
    public void addCart(HttpServletRequest request,HttpServletResponse response) throws Exception {
        //获取页面异步提交过来的要添加的图书id
        String id = request.getParameter("bookId");
        //调用根据id查询图书的方法
        Book book = bookService.findBookById(id);
        /*
          调用购物车对象的方法,往购物车中添加购物项
          但是这里不要直接new Cart(),因为这样的话我们每次点击"添加购物车"都会走addCart方法
          就都会new Cart对象

          我们应该将购物车放到Session中,从Session中获取->定义一个getCart方法
         */
        Cart cart = getCart(request);
        /*
           怎么将购物项添加到购物车呢?
           在Cart类中添加一个方法,专门实现将购物项添加到购物车中
         */
        cart.addCartItem(book);
        //响应客户端:添加购物车成功,异步请求,响应数据都是json
        CommonResult cr = CommonResult.ok().setMessage("添加购物车成功");
        JsonUtils.writeResult(response,cr);

    }

    public Cart getCart(HttpServletRequest request){
        HttpSession session = request.getSession();
        Cart cart = (Cart)session.getAttribute("cart");
        //判断如果Cart对象为空,就将Cart添加到Session域中
        if (cart==null){
            cart = new Cart();
            session.setAttribute("cart",cart);
        }
        return cart;
    }
}

# 2.2.修改Cart类代码

在Cart类中添加购物车
=======================================================
 /**
     * 定义方法，实现将购物项添加到购物车中
     * 传递参数要添加的书对象
     * 添加过程：
     * 1： 判断添加的商品，是不是已经在购物车中了
     * 2： 购物车中存在相同的商品，修改原有购买商品的数量+1
     * 3： 购物车中不存在相同的商品，直接创建购物项对象，存储在购物车中
     * 4： 计算总价格
     */
     /**
     * 添加购物车
     * @param book
     */
    public void addCart(Book book) {
        Integer bookId = book.getBookId();
        if (cartItemMap.containsKey(bookId)){
            CartItem cartItem = cartItemMap.get(bookId);
            cartItem.setCount(cartItem.getCount()+1);
            cartItemMap.put(bookId,cartItem);
        }else{
            CartItem cartItem = new CartItem();
            cartItem.setBookId(bookId);
            cartItem.setBookName(book.getBookName());
            cartItem.setImgPath(book.getImgPath());
            cartItem.setPrice(book.getPrice());
            cartItem.setCount(1);

            cartItemMap.put(bookId,cartItem);

        }

        this.totalAmount+=book.getPrice();
    }

 /*
      首页展示商品总数量
      取出购物车,获取Cart类中的totalCount的值

    public void getTotalCount(HttpServletRequest request,HttpServletResponse response) throws Exception {
        //定义变量，保存购物车的商品数量
        Integer count = 0;
        Cart cart = getCart(request);
        //购物车对象判断
        if(cart != null){
            //调用购物车的方法，获取数量
            count = cart.getTotalCount();
        }
        CommonResult cr = CommonResult.ok().setResultData(count).setMessage("返回购物车总数量");
        JsonUtils.writeResult(response,cr);
    }

# 3.3.修改Cart类

public Integer getTotalCount() {
        /*
          如果每次调用getTotalCount方法
          每次都会在原来的基础上+1
          而购物项是从map中获取的
          那么购物车中有多少购物项,map就遍历几次,就要加几次
          会导致加多了
         */
        this.totalCount = 0;

        Set<Integer> set = cartItemMap.keySet();
        for (Integer key : set) {
            CartItem cartItem = cartItemMap.get(key);
            totalCount+=cartItem.getCount();
        }
        return totalCount;
}

/*
      清空购物车
 */
    public void clearCart(HttpServletRequest request,HttpServletResponse response) throws Exception {
        //从session域中获取Cart
        Cart cart = getCart(request);
        //获取Cart中的Map集合
        Map<Integer, CartItem> map = cart.getCartItemMap();
        //将Map中的CartItem购物项清空
        map.clear();

        //将购物车中的总金额变成0  -> 新添加的
        cart.setTotalAmount(0d);
        CommonResult cr = CommonResult.ok().setMessage("购物车清空成功");
        JsonUtils.writeResult(response,cr);
    }

# 10.解决金额精度损失问题

# 10.1修改Cart类

/**
 * 购物车类
 *  包含，购物项多个，总数据量，购物车所有商品的总价格
 */
public class Cart {
    private Integer totalCount=0;
    private Double totalAmount = 0D;
    //定义大数据运算类
    private BigDecimal totalCountBigDecimal = new BigDecimal("0");

    /**
     * 存储多个购物项的集合
     * Map集合的键，是图书的唯一编号
     * Map集合的值，是购物项
     * 购物项对象CartItem，存储Map集合，添加购物车
     */
    private Map<Integer,CartItem> cartItemMap = new HashMap<Integer, CartItem>();
    public Cart(){

    }
    public Cart(Integer totalCount, Double totalAmount, Map<Integer, CartItem> cartItemMap) {
        this.totalCount = totalCount;
        this.totalAmount = totalAmount;
        this.cartItemMap = cartItemMap;
    }
    /**
     *  移除购物车中的购物项，参数图书id
     *  将Map集合中，移除一组键值对
     *  从新计算总金额
     */
    public void removeCartItem(Integer bookId){
        //Map集合的方法remove()移除键值对，返回被移除之前的值
        CartItem cartItem = cartItemMap.remove(bookId);
        //从新计算总金额
        //总金额 = 原有的总金额 - 被移除的购物项的小计金额
        totalCountBigDecimal = totalCountBigDecimal.subtract(new BigDecimal(cartItem.getAmount()+""));
        this.totalAmount = totalCountBigDecimal.doubleValue();
    }


    /**
     *  修改购买数量的方法，参数，图书id，修改的数量
     *  实现实现：
     *    图书id作为键，取出Map集合的值，购物项CartItem
     *    设置购物项中的购买数量
     *    从新计算总金额
     */
    public void updateCartItemCount(Integer bookId,Integer count){
        CartItem cartItem = cartItemMap.get(bookId);
        //获取原有的购买数量
        Integer originalCount = cartItem.getCount();
        //设置购买数量
        cartItem.setCount(count);
        //从新计算总金额, 原有的总金额 + 购买数量的差值*单价
        totalCountBigDecimal = totalCountBigDecimal.add(new BigDecimal((count-originalCount)*cartItem.getPrice()));
        this.totalAmount = totalCountBigDecimal.doubleValue();
    }
    /**
     * cartItemIncrease 方法，实现购物车中商品数量的++操作
     * 传递的参数：bookId，取出Map集合中的值CartItem 购物项对象
     * 调用购物项对象的方法，实现购买数量+1
     * 从新计算总金额
     */
    public void cartItemIncrease(Integer bookId){
        //购物车对象的Map集合，键获取值，购物项
        CartItem cartItem = cartItemMap.get(bookId);
        // 调用购物项对象的方法，实现购买数量+1
        cartItem.countIncrease();
        //从新计算总金额
        totalCountBigDecimal = totalCountBigDecimal.add(new BigDecimal(cartItem.getPrice()+""));
        this.totalAmount = totalCountBigDecimal.doubleValue();
    }

    /**
     * cartItemDecrease 定义方法，实现购物车中商品数量的--操作
     * 购物车对象：Map集合cartItemMap （书信息，购买的信息，购买数量，小计金额）
     * 实现步骤：
     *   书的唯一编号bookId，Map集合的键，取出Map集合的值 （购物项对象）
     *   取出购物项中的购买数量
     *   判断购买数量是 1 ，Map集合中移除该购物项
     *   判断购买数量是>1 ,购物项的购买数量-1
     *   从新计算购物车的总金额
     *     购物车的金额 = 原有金额 - 购物项的单价*数量
     */
    public void cartItemDecrease(Integer bookId){
        //书的唯一编号bookId，Map集合的键，取出Map集合的值 （购物项对象）
        CartItem cartItem = cartItemMap.get(bookId);
        //调用购物项对象自己的方法，实现购买数量--
        cartItem.countDecrease();
        //判断，购物项，--后，购买数量是不是0
        if (cartItem.getCount() == 0){
            //移除该购物项
            cartItemMap.remove(bookId);
        }
        //计算新的总价格 -= 原有购物项的金额
        totalCountBigDecimal = totalCountBigDecimal.subtract(new BigDecimal(cartItem.getPrice()+""));
        this.totalAmount = totalCountBigDecimal.doubleValue();

       /* //取出购物项中的购买数量
        Integer count = cartItem.getCount();
        //判断数量
        if (count == 1)
            //Map集合中移除该键
            cartItemMap.remove(bookId);
        else {
            //购买数量-1
            cartItem.setCount( cartItem.getCount() - 1 );
        }
*/
    }


    /**
     * 定义方法，实现添加购物车
     * 传递参数要添加的书对象
     * 添加过程：
     *   1： 判断添加的商品，是不是已经在购物车中了
     *   2： 购物车中存在相同的商品，修改原有购买商品的数量+1
     *   3： 购物车中不存在相同的商品，直接创建购物项对象，存储在购物车中
     *   4： 计算总价格
     */
    public void addCartItem(Book book){
        // 1： 判断添加的商品，是不是已经在购物车中了
         //取出书的id值
        Integer bookId = book.getBookId();
        //判断购物车中的Map集合，是否存在此键
        if (cartItemMap.containsKey(bookId)){
            //购物项中，有重复的商品，取出购物项
            CartItem cartItem = cartItemMap.get(bookId);
            //购物项中的购买数量+1
            //2： 购物车中存在相同的商品，修改原有购买商品的数量+1
            cartItem.setCount( cartItem.getCount() + 1);
        }else {
            //购物项集合，没有在购物车中，创建购物项对象
            CartItem cartItem = new CartItem();
            //设置购物项的数据
            cartItem.setBookId(bookId);//书的唯一编号
            cartItem.setBookName( book.getBookName() );//书名
            cartItem.setImgPath( book.getImgPath() );//图片路径
            cartItem.setPrice( book.getPrice() );//单价
            cartItem.setCount(1);//购买数量=1
            //购物项存储Map集合
            //3： 购物车中不存在相同的商品，直接创建购物项对象，存储在购物车中
            cartItemMap.put(bookId,cartItem);
        }
        // 4： 计算总价格， 总价格=要购买的商品的小计金额
        //浮点数据，在内存不精确性
        totalCountBigDecimal =
                totalCountBigDecimal.add(new BigDecimal(book.getPrice()+""));
        //计算后的求和，总价格，赋值变量totalAmount
        totalAmount = totalCountBigDecimal.doubleValue();
    }


    /**
     *
     * 返回购物车中商品的总数据量
     * 进行计算，返回值
     * 相同的商品，算1个，还是单算
     * 购物项对象Cart，添加购物车，就是购物项对象，添加到购物车的Map集合
     * CartItem(购物项)，添加到Cart的Map集合
     *
     * 相同商品算计1个： Map的长度
     * 相同商品单独计算： Map集合的购物项，获取购买数量
     * totalCount 1
     */
    public Integer getTotalCount() {
        this.totalCount = 0;
        //单独计算
        //遍历Map集合
        for(Integer key : cartItemMap.keySet()){
            CartItem cartItem = cartItemMap.get(key);
            //获取购物项中的购买数量
            totalCount += cartItem.getCount();
        }
        return totalCount;
    }

    public void setTotalCount(Integer totalCount) {
        this.totalCount = totalCount;
    }

    public Double getTotalAmount() {
        return totalAmount;
    }

    public void setTotalAmount(Double totalAmount) {
        totalCountBigDecimal = new BigDecimal(0+"");
        this.totalAmount = totalAmount;
    }

    public Map<Integer, CartItem> getCartItemMap() {
        return cartItemMap;
    }

    public void setCartItemMap(Map<Integer, CartItem> cartItemMap) {
        this.cartItemMap = cartItemMap;
    }

    @Override
    public String toString() {
        return "Cart{" +
                "totalCount=" + totalCount +
                ", totalAmount=" + totalAmount +
                ", cartItemMap=" + cartItemMap +
                '}';
    }
}

/**
 *  购物项对象
 *  包含：图书唯一ID，书名，图片，单价，购买数量，小计金额
 */
public class CartItem {
    private Integer bookId;//图书ID
    private String bookName;//图书名
    private String imgPath;//图片路径
    private Double price;//单价
    private Integer count;//购买数量
    private Double amount;//小计金额

    private BigDecimal amountBigDecimal = new BigDecimal("0");

    public CartItem() {
    }

    public CartItem(Integer bookId, String bookName, String imgPath, Double price, Integer count, Double amount) {
        this.bookId = bookId;
        this.bookName = bookName;
        this.imgPath = imgPath;
        this.price = price;
        this.count = count;
        this.amount = amount;
    }
    //购买数量++
    public void countIncrease(){
        this.count++;
    }

    //购买数量 --
    public void  countDecrease(){
        this.count -- ;
    }

    public Integer getBookId() {
        return bookId;
    }

    public void setBookId(Integer bookId) {
        this.bookId = bookId;
    }

    public String getBookName() {
        return bookName;
    }

    public void setBookName(String bookName) {
        this.bookName = bookName;
    }

    public String getImgPath() {
        return imgPath;
    }

    public void setImgPath(String imgPath) {
        this.imgPath = imgPath;
    }

    public Double getPrice() {
        return price;
    }

    public void setPrice(Double price) {
        this.price = price;
    }

    public Integer getCount() {
        return count;
    }

    public void setCount(Integer count) {
        this.count = count;
    }
    //小计金额
    public Double getAmount() {
        BigDecimal bigDecimalCount = new BigDecimal(this.count+"");
        BigDecimal bigDecimalPrice = new BigDecimal(this.price+"");
        this.amount = bigDecimalCount.multiply(bigDecimalPrice).doubleValue();
        return amount;
    }

    public void setAmount(Double amount) {
        this.amount = amount;
    }

    @Override
    public String toString() {
        return "CartItem{" +
                "bookId=" + bookId +
                ", bookName='" + bookName + '\'' +
                ", imgPath='" + imgPath + '\'' +
                ", price=" + price +
                ", count=" + count +
                ", amount=" + amount +
                '}';
    }
}
