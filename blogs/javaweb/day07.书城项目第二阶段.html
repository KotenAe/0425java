      request.getParameter("请求参数的name属性名")

      request.getParameterMap()

    getRequestDispacher(String path).forward(request,response)
    path:写的是跳转的资源路径
  3.域对象:请求域:在一次请求中生效
    setAttribute(String key,Object value):存
    getAttribute(String key):取
    removeAttribute(String key):删除
  4.处理请求中文乱码:
    request.setCharacterEncoding("UTF-8")
  
 2.HttpServletResponse->设置响应信息

     response.setStatus(状态码)

     字符流: getWriter().writer(String str)
     字节流: getOutputStream

    private Integer id;

    private String telephone;

@WebServlet("/register")
public class UserRegisterServlet extends HttpServlet {
    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
        //1.处理请求和响应的中文乱码
        request.setCharacterEncoding("utf-8");
        response.setContentType("text/html;charset=utf-8");
        //2.获取页面上所有提交的请求参数
        Map<String, String[]> map = request.getParameterMap();
        //3.将map中存储的请求参数封装到User对象中
        User user = new User();
        try {
            BeanUtils.populate(user,map);
            //4.调用Dao层的registerUser方法,将封装数据的User对象传递过去
            int result = new UserRegiserDao().registerUser(user);
            //5.根据dao层registerUser方法返回的结果做不同的响应
            if (result==1){
                response.getWriter().write("注册成功");
            }else{
                response.getWriter().write("注册失败");
            }
        } catch (Exception e) {
            e.printStackTrace();
        }
    }

    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
        doGet(request, response);
    }
}


public class UserRegiserDao {
    public int registerUser(User user) throws SQLException {
        //1.创建QueryRunner对象
        QueryRunner qr = new QueryRunner(DruidUtils.getDataSource());
        //2.准备查询的sql
        String find = "select * from t_user where username = ?";
        //3.准备插入的sql
        String add = "insert into t_user values (?,?,?,?,?)";
        //4.先查询,根据用户名查询数据库中有没有用户,如果有(查出数据了)
        List<User> list = qr.query(find, new BeanListHandler<>(User.class), user.getUsername());
        //5.如果查询出来的list中没有元素,证明没查到,可以直接注册
        if (list.size()==0){
            qr.update(add,user.getId(),user.getUsername(),user.getPassword(),user.getEmail(),user.getTelephone());
            //如果注册成功返回一个1
            return 1;
        }else{
            //如果注册失败返回一个0
            return 0;
        }
    }
}

# 2.在HTML中使用base标签统一页面基础访问路径

问题描述:
    1.因为在页面中有很多的a标签、表单以及Ajax请求(以后会学)都需要写访问路径，而在访问路径中项目路径是一样的，所以如果不统一编写项目路径的话，就会发生当项目路径发生改变的时候该页面所有用到项目路径的地方都需要更改的情况

1.base标签作用:给页面上的静态资源路径统一一下路径写法
2.base标签编写位置:必须放到head标签中
3.为什么使用base标签统一一下写法呢:
  因为页面上有各种路径,这些静态资源都放到了web目录下,但是webapp下没有web路径
  项目下直接就是我们引入的各种静态资源,所以我们就不要用./路径写法了,所以我们可以使用base标签定义一个项目名
  让页面上所有静态资源的路径统一变成:项目名/包名/资源名  不用./这种写法了
      
4.写法:
   <base href = ""/>  定义基础的统一访问路径
   href = "写路径",整个页面中的所有路径,都是有href的属性+资源的路径
   
    <base href = "/部署时application context写的名字/"/>

# 第三章.三层架构思想

# 1.为什么要使用三层架构

所有和当前业务功能需求相关的代码全部耦合在一起，如果其中有任何一个部分出现了问题，牵一发而动全身，导致其他无关代码也要进行相应的修改。这样的话代码会非常难以维护。
所以为了提高开发效率，需要对代码进行模块化的拆分。整个项目模块化、组件化程度越高，越容易管理和维护，出现问题更容易排查。
    
用三层架构说白了:
  不同类型的代码放在不同的包中,好维护

# 2.什么是三层架构

1.表现层：又可以称之为控制层，负责处理浏览器请求、返回响应、页面调度
    
2.业务逻辑层：负责处理业务逻辑，根据业务逻辑把持久化层从数据库查询出来的数据进行运算、组装，封装好后返回给表现层，也可以根据业务功能的需要调用持久化层把数据保存到数据库、修改数据库中的数据、删除数据库中的数据
    
3.持久化层：根据上一层的调用对数据库中的数据执行增删改查的操作

public class DruidUtils {
    /**
     * static代码块：读取配置文件 druid.properties
     * 创建出德鲁伊连接池的对象
     */
    private static DataSource dataSource ;
    static {
        try {
            //读取配置文件，类的加载器中的IO流
            InputStream in =
                    DruidUtils.class.getClassLoader().getResourceAsStream("druid.properties");
            //定义集合，存储配置文件的键值对
            Properties prop = new Properties();
            prop.load(in);
            //创建德鲁伊连接池
            dataSource = DruidDataSourceFactory.createDataSource(prop);
        }catch (Exception ex){
            ex.printStackTrace();
            //连接池运行中出现异常，数据库根本就连不上去
            //我们项目中的功能90以上就废弃
            //抛出运行异常，RuntimeException 一旦抛出，Sun公司设计理解，程序人员修改源码
            throw new RuntimeException("数据库连接失败");
        }
    }
    /**
     * 静态方法，返回连接池对象
     */
    public static DataSource getDataSource(){
        return dataSource;
    }
}

# 3.MD5加密

# 3.1.加密方式介绍

1.对称加密：加密和解密使用的相同的密钥，常见的对称加密算法有:DES、3DES
    
2.非对称加密：加密和解密使用的密钥不同，常见的非对称加密算法有:RSA
  加密：使用公钥加密->加密放到服务端
  解密：使用私钥解密->解密放到客户端  -> 只要是自己不把私钥丢了,就可以解密
    
3.消息摘要:  消息摘要算法的主要特征是加密过程不需要密钥，并且经过加密的数据无法被解密，只有相同的原文经过消息摘要算法之后，才能得到相同的密文，所以消息摘要通常用来校验原文的真伪。常用的消息摘要算法有:MD5、SHA、MAC
  abc-> HDFSDDFADFAFA  -> 注册进去的
  abc-> HDFSDDFADFAFA  ->登录填写的
    
4.我们在书城项目中采用MD5算法对密码进行加密,而MD5加密算法是jdk自带的

# 3.2.MD5加密工具类

public class MD5Utils {

    /**
     * 针对明文字符串执行MD5加密
     * @param source
     * @return
     */
    public static String encode(String source) {

        // 1.判断明文字符串是否有效
        if (source == null || "".equals(source)) {
            throw new RuntimeException("用于加密的明文不可为空");
        }

        // 2.声明算法名称
        String algorithm = "md5";

        // 3.获取MessageDigest对象
        MessageDigest messageDigest = null;
        try {
            messageDigest = MessageDigest.getInstance(algorithm);
        } catch (NoSuchAlgorithmException e) {
            e.printStackTrace();
        }

        // 4.获取明文字符串对应的字节数组
        byte[] input = source.getBytes();

        // 5.执行加密
        byte[] output = messageDigest.digest(input);

        // 6.创建BigInteger对象
        int signum = 1;
        BigInteger bigInteger = new BigInteger(signum, output);

        // 7.按照16进制将bigInteger的值转换为字符串
        int radix = 16;
        String encoded = bigInteger.toString(radix).toUpperCase();

        return encoded;
    }
}

public interface UserDao {
    int registerUser(User user) throws SQLException;
}

public class UserDaoImpl implements UserDao {
    @Override
    public int registerUser(User user) throws SQLException {
        //1.创建QueryRunner对象
        QueryRunner qr = new QueryRunner(DruidUtils.getDataSource());
        //2.准备查询语句
        String find = "select * from t_user where user_name = ?";
        //3.准备插入语句
        String add = "insert into t_user values (?,?,?,?)";
        //4.先查询
        List<User> list = qr.query(find, new BeanListHandler<>(User.class), user.getUsername());
        //5.判断如果集合长度==0,证明根据用户名没查到,证明没注册过,就执行添加,进行注册
        if (list.size() == 0) {
            qr.update(add, user.getUserid(), user.getUsername(), user.getPassword(), user.getEmail());
            //返回1,证明注册成功
            return 1;
        } else {
            //返回0,证明注册失败
            return 0;
        }
    }
}

        request.setCharacterEncoding("utf-8");
        response.setContentType("text/html;charset=utf-8");
        //2.获取页面上提交的参数

        String password = request.getParameter("password");

        //4.调用service层的loginUser,将username,和password传递过去
        UserService userService = new UserServiceImpl();

            User user = userService.loginUser(username, password);

              注意:dao层那边根据username,password能查出user对象来
