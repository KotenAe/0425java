   th:text = "${域中的key}"
 3.设置标签的value:
   th:value = "${域中的key}"
 4.分支语句
   th:if = "结果为true"  -> 显示当前标签
   th:if = "结果为false" -> 不显示当前标签
   th:unless = "结果为true" -> 不显示当前标签
   th:unless = "结果为false" -> 显示当前标签
 5.遍历:
   th:each = "变量名:${域中的key}"
 6.获取请求参数:
   ${param.属性名}
 7.获取域中的数据
   a.request:   ${域中的key}
   b.session:   ${session.域中的key}
   b.servletContext:  ${application.域中的key}
 8.内置对象:lists
   ${#lists.isEmpty(域中的key)}
 9.OGNL表达式:基于剥洋葱的思想
 10.想使用Thymeleaf:
    a.导jar包
    b.在web.xml中配置前缀(前面的路径)和后缀(文件后缀名)
    c.使用Thymeleaf模板(工具类)

由于我们将书城的首页index.html放到了WEB-INF下,浏览器直接访问不了了,所以我们就需要一个Servlet转发到WEB-INF下的index.html

/*
  注意:
    此Servlet的url-pattern配置成首页,然后跳转到WEB-INF下的首页index.html
    由于tomcat启动会先访问index.html,WEB目录下没有(因为index.html放到WEB-INF下了)
    但是Servlet中写了index.html,所以tomcat就会去访问Servlet中设置的index.html
    也就访问了此Servlet
    
    以后写localhost:8080/bookstore
         localhost:8080/bookstore/index.html
    都会跳转到这个Servlet中,也就转发到了WEB-INF下的首页index.html中
 */
@WebServlet("/index.html")
public class PortalServlet extends ViewBaseServlet {
    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
        processTemplate("index",request,response);
    }

    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
        doGet(request, response);
    }
}

 //用户登录
    public void doLogin(HttpServletRequest request, HttpServletResponse response) throws IOException {
        //获取页面提交的参数
        String username = request.getParameter("username");
        String password = request.getParameter("password");
        //加密,因为数据库中存进去的数据都是加密的,所以只有加密之后才能和数据库的数据对上
        String pwd = MD5Utils.encode(password);
        try {
            //调用Service层方法
            User user = service.loginUser(username, pwd);
            if (user!=null){
                processTemplate("user/login_success",request,response);
            }else{
                //将错误提示放到域中
                request.setAttribute("loginMessage","对不起,登录失败!");
                //将页面重新跳转到登录页面
                processTemplate("user/login",request,response);
            }
        } catch (Exception e) {
            e.printStackTrace();
        }
    }

    public void saveOrUpdateBook(HttpServletRequest request, HttpServletResponse response) throws IOException {
        //获取页面提交的参数
        Map<String, String[]> map = request.getParameterMap();
        //创建Book对象
        Book book = new Book();
        try{
            BeanUtils.populate(book,map);
            //调用service层添加和修改方法
            bookService.saveOrUpdateBook(book);
            //页面重定向到"图书管理页面"
            response.sendRedirect(request.getContextPath()+"/admin?method=toBookManager");
        }catch (Exception e){
            e.printStackTrace();
        }
    }

# 5.3.2.Service层

//接口->不改代码
    int saveOrUpdateBook(Book book) throws SQLException;
=========================================================
//实现类->改代码
      /*
      做个判断
      如果是修改操作,肯定带id,id不为null,因为需要根据id修改图书
      如果是添加操作,肯定id就为null,id为主键自增,肯定不用带id
     */
    @Override
    public int saveOrUpdateBook(Book book) throws SQLException {
        Integer id = book.getBookId();
        if (id==null){
            //固定设置图书的图片路径,因为没有做文件上传,所以把添加的图片写死
            book.setImgPath("static/uploads/jiaofu.jpg");
            int result01 = bookDao.addBook(book);
            return result01;
        }else{
            int result02 = bookDao.updateBook(book);
            return result02;
        }
    }

# 1.前台图书展示服务端实现

1.我们访问index.html其实是访问的PortalServlet,因为我们在PortalServlet中设置了index.html.所以我们访问的书城首页其实是   通过PortalServlet转到index.html书城首页的
2.我们在首页展示的图书其实就是数据库中的数据,而查询数据库中所有的图书功能,我们已经在做"图书管理"的时候已经做完了,所以我们可以   在PortalServlet中直接调用BookService中的showBookList方法.

# 1.1.Servlet层_PortalServlet

@WebServlet("/index.html")
public class PortalServlet extends ViewBaseServlet {
    private BookService bookService = new BookServiceImpl();
    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
        try {
            List<Book> list = bookService.showBookList();
            //将list存到request域中
            request.setAttribute("booklist",list);
            processTemplate("index",request,response);
        } catch (SQLException e) {
            e.printStackTrace();
        }
    }

    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
        doGet(request, response);
    }
}
