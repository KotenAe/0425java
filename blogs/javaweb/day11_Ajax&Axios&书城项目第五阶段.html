# 4.同步和异步

# 4.1.Ajax介绍

   Ajax本身就是Asynchronous JavaScript And XML的缩写，直译为：异步的[JavaScript]和[XML]。在实际应用中Ajax指的是不刷新浏览器窗口,不做页面跳转,局部更新页面内容的技术。  
    
   以前ajax的数据放到xml中,现在都是json数据格式存放了  {"username":"root","password":"1234"}

# 4.2.同步和异步区别

1.同步:多个操作按顺序执行,前面的操作没有完成,后面的操作必须等待
2.异步:多个操作相继并发执行,即使开始的先后顺序不同,但是由于他们各自在自己独立的进程或者线程中完成,所以互不干扰

# 3.2.axios处理响应数据

1.语法1:
  axios({
     method:"post",//请求方式
     url:"/web项目名/servlet的url-pattern",  //请求路径
     params:{    //请求参数
       "key":value
     }
  }).then(function(response){//回调函数
     处理响应数据
  }).catch(function(response){//回调函数
     处理响应数据
  })
=========================================================
2.语法2:简化形式
    axios({
     method:"post",//请求方式
     url:"/web项目名/servlet的url-pattern",  //请求路径
     params:{    //请求参数
       "key":value
     }
    }).then(response=>{
       处理响应数据
    }).catch(response=>{
       处理响应数据
    })
    
=========================================================
3.细节说明:
  a.如果响应成功了走then,响应失败了走catch,但是一般我们只关注响应成功,只写then
      
  b.then和catch后面的function叫做回调函数(是我们写的,但不是我们调用的,是axios对象调用的),表示响应成功或者响应失败要执行的函数,在函数里面写响应数据的处理方式
      
  c.function中的response就是一个变量名,名字随意写,只不过习惯上使用response
    response接收的就是服务器响应回来的数据
      
  d.function可以省略,只留下function中的参数,类似于lambda表达式
    写成:  变量=>{回调函数的方法体}
    前提:  使用这种省略格式需要将ECMA版本调成6版本才行
        
  e.response这个变量接收的响应数据会有很多,但是其中我们只关注"data"即可
    data接收的数据才是我们想要的响应数据
    获取data:  response.data

@WebServlet("/axios4")
public class Axios04Servlet extends HttpServlet {
    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
       //request调用getReader方法,获取BufferedReader对象->读取axios的请求参数
        BufferedReader br = request.getReader();
        String line;

        StringBuilder sb = new StringBuilder();
        while((line = br.readLine())!=null){
            //拼接字符串,如果发送过来的json数据是个数组也照样拼接
            sb.append(line);
        }
        //将sb拼接的内容变成字符串
        String json = sb.toString();
        //将前端发送过来的json数据封装成对象
        ObjectMapper objectMapper = new ObjectMapper();
        Person person = objectMapper.readValue(json, new TypeReference<Person>() {
        });
        System.out.println(person);
    }

    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
        doGet(request, response);
    }
}

public class Person {
    private String username;
    private String password;

    public Person() {
    }

    public Person(String username, String password) {

    }

    public String getUsername() {
        return username;
    }

    public void setUsername(String username) {
        this.username = username;
    }

    public String getPassword() {
        return password;
    }

    public void setPassword(String password) {
        this.password = password;
    }

    @Override
    public String toString() {
        return "Person{" +
                "username='" + username + '\'' +
                ", password='" + password + '\'' +
                '}';
    }
}

public class Axios05Servlet extends HttpServlet {
    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
       //request调用getReader方法,获取BufferedReader对象->读取axios的请求参数
        BufferedReader br = request.getReader();
        String line;

        StringBuilder sb = new StringBuilder();
        while((line = br.readLine())!=null){
            //拼接字符串,如果发送过来的json数据是个数组也照样拼接
            sb.append(line);
        }
        //将sb拼接的内容变成字符串
        String json = sb.toString();
        //将前端发送过来的json数据封装成对象
        ObjectMapper objectMapper = new ObjectMapper();
        Person person = objectMapper.readValue(json, new TypeReference<Person>() {
        });

        //将封装好的Person对象再响应给axios
        response.setContentType("text/html;charset=utf-8");
        String s = objectMapper.writeValueAsString(person);
        response.getWriter().write(s);
    }

    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
        doGet(request, response);
    }
}

public class JsonUtils {
    private static ObjectMapper mapper = new ObjectMapper();
    public static Object parseJsonBean(HttpServletRequest request,Class clazz){
        BufferedReader br = null;
        try {
            //request调用getReader方法,获取BufferedReader对象
            br = request.getReader();
            String line;
            StringBuilder sb = new StringBuilder();
            while((line = br.readLine())!= null){
                sb.append(line);
            }
            String json = sb.toString();
            //System.out.println(json);//{"username":"tom","password":"1234"}
            Object object= mapper.readValue(json, clazz);
            return object;
        } catch (Exception e) {
            e.printStackTrace();
            throw new RuntimeException(e.getMessage());
        } finally {
            try {
                br.close();
            } catch (IOException e) {
                e.printStackTrace();
                throw new RuntimeException(e.getMessage());
            }
        }
    }

    //将对象转成json字符串响应给客户端
    public static void writeResult(HttpServletResponse response, Object obj){
        try {
            response.setContentType("text/html;charset=utf-8");
            //将user响应到页面上
            response.getWriter().write(mapper.writeValueAsString(obj));
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}

在整个项目中，凡是涉及到给Ajax请求设置响应，我们都封装到CommonsResult类型中。到时候想要给客户端设置响应数据,我们直接响应此类对象即可

/**
 * 包名:com.atguigu.bean
 *
 * @author Leevi
 * 日期2021-05-21  10:43
 
 
 作用:设置响应信息的
     将响应信息封装到此对象中,再将此对象转成json响应给浏览器
     浏览器就可以直接获取到此对象中封装的响应信息
 */
public class CommonResult {
    /**
     * 服务器端处理请求的标示
     
       表示服务器处理请求的状态,成功为true 失败为false
       比如登录成功为true   登录失败false
     */
    private boolean flag;

    /**
     * 当服务器端处理请求成功的时候要显示给客户端的数据(主要针对于查询)
     */
    private Object resultData;
    
    /**
     * 当服务器端处理请求失败的时候要响应给客户端的错误信息
       
       比如登录失败时要显示什么错误信息
     */
    private String message;

    /**
     * 处理请求成功
     * @return
     */
    public static CommonResult ok(){
        return new CommonResult().setFlag(true);
    }

    /**
     * 处理请求失败
     * @return
     */
    public static CommonResult error(){
        return new CommonResult().setFlag(false);
    }

    public boolean isFlag() {
        return flag;
    }

    private CommonResult setFlag(boolean flag) {
        this.flag = flag;
        return this;
    }

    public Object getResultData() {
        return resultData;
    }

    //当服务器端处理请求成功的时候要显示给客户端的数据(主要针对于查询)
    public CommonResult setResultData(Object resultData) {
        this.resultData = resultData;
        return this;
    }

    public String getMessage() {
        return message;
    }

    public CommonResult setMessage(String message) {
        this.message = message;
        return this;
    } 

    @Override
    public String toString() {
        return "CommonResult{" +
                "flag=" + flag +
                ", resultData=" + resultData +
                ", message='" + message + '\'' +
                '}';
    }
}

            //校验用户名
            checkUsername(){
                //注册的用户名功能必须是3-16位的数字或者大小写字母
                var regexName = /^[0-9A-Za-z]{3,16}$/;
                /*
                  判断输入的内容是否符合用户的正则
                  如果输入的用户名符合正则表达式再发起异步请求
                  如果进了if,证明正则通过了,在if中写axios代码
                 */
                if (regexName.test(this.username)){
                    axios({
                        method:"post",
                        url:"user",
                        params:{
                            //提交的参数中指明要调用的方法以及要校验的参数
                            /*
                              localhost:8080/bookstore/user?method=checkUsername&username=tom
                            */
                            "method":"checkUsername",
                            "username":this.username
                        }
                    }).then(response=>{
                        if (response.data.flag==true){
                            this.usernameMessage = "";
                            return true;
                        }else{
                            this.usernameMessage = response.data.message;
                            return false;
                        }
                    })

                } else{
                    //否则要有提示,证明不符合正则
                    this.usernameMessage = "用户名格式不正确";
                    return false;
                }
            }
