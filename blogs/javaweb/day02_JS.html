# 第一章.JavaScript概述和体验

# 1.什么是JavaScript

1.概述:JavaScript是一种属于网络的脚本语言(需要嵌套在html页面中使用的),用于web开发
2.作用:给网页添加一些动态功能,提高用户体验,为用户提供了更加流畅美观的浏览效果

# 2.JS三大组成部分

组成部分作用
ECMA Script所有脚本语言规范,构成了js的语法基础
BOMBrowser Object Model 浏览器对象模型,用来操作浏览器中各个对象
DOMDocument Object Model 文档对象模型,用来操作网页中的各个元素(标签)
获取标签 操作标签属性(获取属性的值, 或者为属性赋值) 操作标签体的文本部分

# 3.网页中各技术的作用

技术作用
HTML创建网页,构建网页
CSS美化网页
JavaScript用于网页交互,让网页动起来,让网页更加美观,提交用户体验

数字可以与字符串进行比较，字符串可以与字符串进行比较。字符串与数字进行比较的时候会先把字符串转换成数字然后再进行比较

# 第九章.JS函数

# 1.函数介绍

1.什么是函数:
  一个代码块,给代码取个名字,后面可以用过名字来调用它,实现复用的功能,类似于java中的方法(方法就是功能)
2.js中有两种函数:
  命名函数:带名字的函数,可以用过名字来多次调用
  匿名函数:不带名字的函数,不可重用,每次使用一次,后面常用于和js事件结合使用

# 2.函数定义方式1_命名函数

1.格式:
  function 名字(参数){
    方法体
    return 返回值;
  }
2.注意:
  a.js中的函数不用写返回值类型
  b.js中的函数参数位置直接写变量名,连var let都不用写
  c.js中的函数有没有返回值,取决于函数内部写没写return

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>命名函数</title>
</head>
<body>

</body>
<script>
    /*
      1.格式:
        function 名字(参数){
          方法体
          return 返回值;
        }
      2.注意:
        a.js中的函数不用写返回值类型
        b.js中的函数参数位置直接写变量名,连var let都不用写
        c.js中的函数有没有返回值,取决于函数内部写没写return
     */
    //无参无返回值
    function method01(){
        console.log("无参无返回值的函数");
    }
    method01();
    //有参无返回值
    function method02(a,b){
        console.log(a+b);
    }
    method02("张无忌","张三丰");
    //无参有返回值
    function method03(){
        return "涛哥是尚硅谷第一帅";
    }
    var result01 = method03();
    console.log(result01);
    //有参有返回值

    function method04(a,b){
        return a+b;
    }

    var result02 = method04(10,20);
    console.log(result02);
</script>
</html>

# 1.DOM对象_查找标签方法

document.getElementById("id")   根据标签中的id值获取标签对象,由于id要求唯一,所以返回单个标签对象
document.getElementsByTagName("标签名")  根据标签名,获取一组标签,返回的是数组
document.getElementsByName("name值")  根据标签中的name的属性值获取标签对象,返回数组
document.getElementsByClassName("类名")根据标签中的class的值获取一组标签,返回的是数组

<head>
    <meta charset="UTF-8">
    <title>获取标签对象</title>

<body>

<div>根据标签名获取标签对象1</div>
<div>根据标签名获取标签对象2</div>

<div name="divName">根据name值获取标签对象1</div>
<div name="divName">根据name值获取标签对象2</div>

<div class="divClass">根据class值获取标签对象1</div>
<div class="divClass">根据class值获取标签对象2</div>
</body>
<script>
    //document.getElementById("id")   根据标签中的id值获取标签对象,由于id要求唯一,所以返回单个标签对象
    var elementId = document.getElementById("d1");
    console.log(elementId);
    console.log("==================================");
    //document.getElementsByTagName("标签名")  根据标签名,获取一组标签,返回的是数组
    var elementsByTagName = document.getElementsByTagName("div");
    for (var i = 0; i < elementsByTagName.length; i++) {
        console.log(elementsByTagName[i]);
    }
    console.log("==================================");
    //document.getElementsByName("name值")  根据标签中的name的属性值获取标签对象,返回数组
    var elementsByName = document.getElementsByName("divName");
    for (var i = 0; i < elementsByName.length; i++) {
        console.log(elementsByName[i]);
    }
    console.log("==================================");
    //document.getElementsByClassName("类名")根据标签中的class的值获取一组标签,返回的是数组
    var elementsByClassName = document.getElementsByClassName("divClass");
    for (var elementClass of elementsByClassName) {
        console.log(elementClass);
    }
</script>
</html>

# 3.标签体操作

需求操作方式

innerText和innerHTML区别:
1.相同点:
  两个都能获取标签的标签体,都可以设置标签的标签体
2.不同点:
   innerText:不能让标签生效
   innerHTML:能让标签生效

    var elementById2 = document.getElementById("s2");

    //为第三个div文本部分设置内容
    var elementById3 = document.getElementById("s3");

      innerText和innerHTML区别:

        两个都能获取标签的标签体,都可以设置标签的标签体

    var elementById4 = document.getElementById("s4");

# 4.标签添加和删除

API功能
document.createElement("标签名")创建元素节点并返回被创建的标签对象，但不会自动添加到文档中,需要用到appendChild函数将创建的标签添加到指定的标签中
标签对象.appendChild(ele)将ele添加到标签对象所有子节点后面
parentEle.insertBefore(newEle,targetEle)将newEle插入到targetEle前面
parentEle.replaceChild(newEle, oldEle)用新节点替换原有的旧子节点
element.remove()删除某个标签,element指的是要删除的标签对象
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>添加和删除标签</title>
</head>
<body>
<ul id="city">
    <li id="bj">北京</li>
    <li id="sh">上海</li>

    <li id="sz">深圳</li>
    <li id="gz">广州</li>
</ul>
</body>

<script>
    /*
     需求1:创建li标签,标签体为"西安"
          将li标签放到ul标签中
    */
    //先创建li标签
    var liElement = document.createElement("li");
    //给li标签设置标签体
    liElement.innerText = "西安";
    //将设置好的li标签添加到ul标签中
    var ulElement = document.getElementById("city");
    ulElement.appendChild(liElement);

    /*
     需求2:将"西安"添加到深圳前面
    */
    var szElement = document.getElementById("sz");
    ulElement.insertBefore(liElement,szElement);

    /*
    需求3:用"西安"替换"深圳"
    */
    ulElement.replaceChild(liElement,szElement);

    /*
     需求4:删除上海这个标签
    */
    var shElement = document.getElementById("sh");
    shElement.remove();

</script>
</html>

# 第十一章_JS事件

# 1.JS事件介绍

用户再操作网页的时候,会有各种操作,每种操作都会有一个对应的事件,我们可以通过js来监听这些事件,对这些事件进行编写函数来处理具体的操作,来实现与用户的交互
    
事件可以理解为监听用户在页面上做的动作,用户做完这些动作,页面会出现什么样的效果呢?需要将整个事件绑定一个函数来处理
    
    
简单理解:
   1.事件监听的是用户在页面上的动作,操作
   2.事件监听到了用户做的动作,应该具体做写什么回应给用户?这时就需要将事件去绑定一个函数,让这个函数做具体的回应

# 2.设置JS事件的两种方式

1.使用命名函数绑定事件
  a.直接在标签中使用-->  on事件名 = "函数名()"
  b.比如:
    <input type = "button" onclick = "sum()"/>
    function sum(){
        事件具体的处理方式
    }


2.匿名函数绑定事件
  a.格式:标签对象.on事件名 = function(){
          事件具体的处理方式
         }
  b.比如:
    <input type = "button" id = "b1"/>
    document.getElementById("b1").onclick = function(){
        事件具体的处理方式 
    }

    document.getElementById("b1").onclick->代表的是给指定的标签绑定一个指定的单击事件


<!DOCTYPE html>
<html lang="en">
<head>

        console.log("我是命名函数绑定的事件");

    //获取标签对象

        console.log("我是匿名函数绑定的事件");
    }

# 第十二章.JSON

# 1.Json介绍

(JavaScript] Object Notation) 是一种轻量级的数据交换格式。它基于 ECMAScript (欧洲计算机协会制定的js规范)的一个子集，采用完全独立于编程语言的文本格式来存储和表示数据。简洁和清晰的层次结构使得 JSON 成为理想的数据交换格式。 易于人阅读和编写，同时也易于机器解析和生成，并有效地提升网络传输效率。

# 2.Json的数据格式

1.数组格式：[obj,obj,obj...]，使用中括号包裹，数组的元素可以是任意类型，多个元素之间逗号分开。

2.对象格式：{"key1":obj,"key2":obj,"key3":obj...}，使用大括号包裹，对象采用键值对形式，键必须是字符串类型，值可以是任意类型，多个键值对之间逗号分开。

3.数组和对象之间可以相互嵌套

    //遍历数组
    for (var objectValue of valueArray) {
        console.log(objectValue.name+"..."+objectValue.age);
    }
</script>

# 3.6.JSON和字符串之间的转换

# 3.6.1.json转成string
var jsonObj = {"stuName":"tom","stuAge":20};
var jsonStr = JSON.stringify(jsonObj);

console.log(typeof jsonObj); // object
console.log(typeof jsonStr); // string
# 3.6.2.string转成json
jsonObj = JSON.parse(jsonStr);
