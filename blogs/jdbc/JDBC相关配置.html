# JDBC相关配置

# 前言

所有的java代码操作数据都是在JDBC基础上演化过来的
所有的java代码操作数据库的技术都需要共同的jar包:
mysql-connector-java-8.0.25.jar   -> url中不用加时区
或者
mysql-connector-java-8.0.19.jar

# 一.使用原生JDBC操作数据库

# 1.需要的jar包

1.导jar包:
  mysql-connector-java-8.0.25.jar   -> url中不用加时区
  或者
  mysql-connector-java-8.0.19.jar

# 2.需要的工具类

public class JDBCUtils01 {
    private static String url;
    private static String user;
    private static String password;

    //利用静态代码块让注册驱动,准备获取连接的参数最先执行
    static{
        try {
             Class.forName("com.mysql.cj.jdbc.Driver");
             url = "jdbc:mysql://localhost:3306/220212_java3?serverTimezone=UTC&rewriteBatchedStatements=true";
             user = "root";
             password = "root";
        } catch (ClassNotFoundException e) {
            e.printStackTrace();
        }
    }


    public static Connection getConnection()throws Exception{
        /*
           2.获取连接
             Connection接口
             a.获取:Connection conn = DriverManager.getConnection(String url, String user, String password)
             b.参数:
               url:数据库的地址 -> jdbc:mysql://localhost:3306/数据库名字
               user:数据库的用户名->root
               password:数据库的密码->自己安装mysql设置的密码
         */

        Connection connection = DriverManager.getConnection(url, user, password);

        return connection;
    }

    //关闭资源
    public static void close(Connection connection, Statement statement, ResultSet resultSet){
        if (resultSet!=null){
            try {
                resultSet.close();
            } catch (SQLException e) {
                e.printStackTrace();
            }
        }


        if (statement!=null){
            try {
                statement.close();
            } catch (SQLException e) {
                e.printStackTrace();
            }
        }

        if (connection!=null){
            try {
                connection.close();
            } catch (SQLException e) {
                e.printStackTrace();
            }
        }
    }
}

# 3.原生jdbc读取配置文件方式

# 3.1配置文件内容

1.在resources资源包下创建properties文件

2.具体配置:
driverClassName=com.mysql.cj.jdbc.Driver
url=jdbc:mysql://localhost:3306/220212_java3?serverTimezone=UTC&rewriteBatchedStatements=true
username=root
password=root

# 3.2读取配置文件工具类

public class JDBCUtils02 {
    private static String driverClassName;
    private static String url;
    private static String user;
    private static String password;
    static{
       try{
           //创建Properties对象
           Properties properties = new Properties();
           //利用InputStream读取配置文件
           InputStream is = JDBCUtils02.class.getClassLoader().getResourceAsStream("mysql.properties");
           //利用load方法将流中的信息加载到Properties中
           properties.load(is);
           //调用getProperty(String key)根据key获取value
           driverClassName = properties.getProperty("driverClassName");
           Class.forName(driverClassName);

           url = properties.getProperty("url");
           user = properties.getProperty("username");
           password = properties.getProperty("password");
       }catch (Exception e){
           e.printStackTrace();
       }
    }

    public static Connection getConnection()throws Exception{
        /*
           2.获取连接
             Connection接口
             a.获取:Connection conn = DriverManager.getConnection(String url, String user, String password)
             b.参数:
               url:数据库的地址 -> jdbc:mysql://localhost:3306/数据库名字
               user:数据库的用户名->root
               password:数据库的密码->自己安装mysql设置的密码
         */

        Connection connection = DriverManager.getConnection(url, user, password);

        return connection;
    }

    //关闭资源
    public static void close(Connection connection, Statement statement, ResultSet resultSet){
        if (resultSet!=null){
            try {
                resultSet.close();
            } catch (SQLException e) {
                e.printStackTrace();
            }
        }


        if (statement!=null){
            try {
                statement.close();
            } catch (SQLException e) {
                e.printStackTrace();
            }
        }

        if (connection!=null){
            try {
                connection.close();
            } catch (SQLException e) {
                e.printStackTrace();
            }
        }
    }
}

# 二.使用C3P0连接池操作数据库

# 1.需要的jar包

c3p0-0.9.1.2.jar

# 2.需要的配置文件

1.在resources资源包下创建xml文件,文件名字固定:c3p0-config.xml
2.详细配置:
  <?xml version="1.0" encoding="UTF-8"?>
<c3p0-config>
    <!-- 使用默认的配置读取连接池对象 -->
    <default-config>
        <!--  连接参数 -->
        <property name="driverClass">com.mysql.cj.jdbc.Driver</property>
        <property name="jdbcUrl">jdbc:mysql://localhost:3306/220212_java3?serverTimezone=UTC&rewriteBatchedStatements=true</property>
        <property name="user">root</property>
        <property name="password">root</property>

        <!-- 连接池参数 -->
        <property name="initialPoolSize">5</property>
        <property name="maxPoolSize">10</property>
        <property name="checkoutTimeout">2000</property>
        <property name="maxIdleTime">1000</property>
    </default-config>
</c3p0-config>

# 3.需要的工具类

public class C3P0Utils {
    //定义连接池对象->DataSource实现类->ComboPooledDataSource
    private static ComboPooledDataSource cpds;
    static{
        cpds = new ComboPooledDataSource();
    }

    //获取连接
    public static Connection getConnection()throws Exception{
        return cpds.getConnection();
    }

    //关闭资源
    public static void close(Connection connection, Statement statement, ResultSet resultSet){
        if (resultSet!=null){
            try {
                resultSet.close();
            } catch (SQLException e) {
                e.printStackTrace();
            }
        }


        if (statement!=null){
            try {
                statement.close();
            } catch (SQLException e) {
                e.printStackTrace();
            }
        }

        if (connection!=null){
            try {
                connection.close();
            } catch (SQLException e) {
                e.printStackTrace();
            }
        }
    }
}

# 三.使用Druid连接池操作数据库

# 1.需要的jar包

druid-1.1.6.jar

# 2.需要的配置文件

1.在resources资源包下创建druid.properties文件

2.详细配置
driverClassName=com.mysql.cj.jdbc.Driver
url=jdbc:mysql://localhost:3306/220212_java3?serverTimezone=UTC&rewriteBatchedStatements=true
username=root
password=root

# 3.需要的工具类

public class DruidUtils {
    private static DataSource ds;
    static{
        try{
            //创建Properties文件
            Properties properties = new Properties();
            //读取配置文件
            InputStream is = DruidUtils.class.getClassLoader().getResourceAsStream("druid.properties");
            //load方法将流中的数据加载到Properties集合中
            properties.load(is);

            //DruidDataSourceFactory.createDataSource(prop);->根据Propertis集合中的数据创建连接池对象

            ds = DruidDataSourceFactory.createDataSource(properties);
        }catch (Exception e){
            e.printStackTrace();
        }

    }

    //获取连接
    public static Connection getConnection() throws Exception {
        Connection connection = ds.getConnection();
        return connection;
    }

    //关闭资源
    public static void close(Connection connection, Statement statement, ResultSet resultSet){
        if (resultSet!=null){
            try {
                resultSet.close();
            } catch (Exception e) {
                e.printStackTrace();
            }
        }

        if (statement!=null){
            try {
                statement.close();
            } catch (Exception e) {
                e.printStackTrace();
            }
        }

        if (connection!=null){
            try {
                connection.close();
            } catch (Exception e) {
                e.printStackTrace();
            }
        }

    }
}
